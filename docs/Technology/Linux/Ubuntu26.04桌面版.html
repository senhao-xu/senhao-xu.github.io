安装输入法 ​

bash
apt update && sudo apt install fcitx5 fcitx5-chinese-addons
im-config -n fcitx5

删除Ibus(可选) ​

bash
apt purge --auto-remove ibus

添加开机自启动 ​

bash
mkdir -p ~/.config/autostart

cat > ~/.config/autostart/fcitx5.desktop <<EOF
[Desktop Entry]
Type=Application
Name=Fcitx5
Comment=Start Fcitx 5 Input Method
Exec=fcitx5 -d
X-GNOME-Autostart-enabled=true
NoDisplay=true
EOF

关闭memory test引导 ​

bash
# 编辑
vim /etc/default/grub
# 添加
GRUB_DISABLE_MEMTEST=true
# 更新引导
update-grub

关闭自动更新 ​

  • 关闭自动更新
bash
sudo sed -i 's/"1"/"0"/g' /etc/apt/apt.conf.d/20auto-upgrades
  • 禁用无人值守升级服务
bash
systemctl stop unattended-upgrades && systemctl disable unattended-upgrades
  • 禁用定时检查更新的任务
bash
systemctl stop apt-daily.timer && systemctl disable apt-daily.timer
systemctl stop apt-daily-upgrade.timer && systemctl disable apt-daily-upgrade.timer

安装Idea ​

bash
# 解压文件
sudo tar -zxvf ideaIU-2026.1.3.tar.gz -C /home/xusenhao/software/idea

# 一般运行完./idea.sh后会自动生成图标无需执行添加图标，没有再执行

# 添加图标
cat << EOF > ~/.local/share/applications/idea.desktop
[Desktop Entry]
Name=IntelliJ IDEA Community
Comment=Java IDE by JetBrains
Exec=/home/用户名/idea/idea-IC-*/bin/idea.sh
Icon=/home/用户名/idea/idea-IC-*/bin/idea.png
Terminal=false
Type=Application
Categories=Development;IDE;
StartupWMClass=jetbrains-idea
EOF

安装java ​

bash
# 创建目录
sudo mkdir -p /usr/lib/java
# 解压文件
sudo tar -zxvf jdk-8u391-linux-x64.tar.gz
# 迁移
sudo mv jdk1.8.0_391 /usr/lib/java/
# 配置环境变量
cat >> ~/.bashrc <<EOF
export JAVA_HOME=/usr/lib/java/jdk1.8.0_391
export PATH=$JAVA_HOME/bin:$PATH
EOF
# 生效
source ~/.bashrc
# 验证
java -version

挂载Windows磁盘 ​

  • 临时挂载
bash
# 查看磁盘分配
sudo fdisk -l
# 找到需要挂载的Windows分区，例如/dev/nvme0n1p3，
# 并确认其文件系统类型（通常为NTFS）。然后，创建一个挂载点，例如/mnt/C
sudo mkdir /mnt/c 
# 执行挂载命令
sudo mount -t ntfs /dev/nvme0n1p3 /mnt/C -o iocharset=utf8,umask=0
# 卸载
sudo umount /dev/nvme0n1p3
  • 永久挂载
bash
# 编辑fstab
sudo vim /etc/fstab
# 挂载
/dev/nvme0n1p3 /mnt/c ntfs defaults 0 0
# 测试
sudo mount -a
