制作Kubernetes离线资源包 ​

依赖包 ​

bash
yum install -y yum-utils
yum install -y yum-plugin-downloadonly

Docker ​

yum源 ​

bash
yum-config-manager --add-repo https://download.docker.com/linux/centos/docker-ce.repo
bash
yum-config-manager --add-repo http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo

查看可用版本 ​

bash
yum list docker-ce --showduplicates | sort -r

下载离线包 ​

bash
yum install -y --downloadonly --downloaddir=rpm/docker-rpm docker-ce-20.10.9-3.el7 docker-ce-cli-20.10.9-3.el7 containerd.io docker-compose-plugin

Kubernetes ​

yum源 ​

bash
cat <<EOF | sudo tee /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://pkgs.k8s.io/core:/stable:/v1.33/rpm/
enabled=1
gpgcheck=1
gpgkey=https://pkgs.k8s.io/core:/stable:/v1.33/rpm/repodata/repomd.xml.key
EOF
bash
cat > /etc/yum.repos.d/kubernetes.repo << EOF
[kubernetes]
name=Kubernetes
baseurl=https://mirrors.aliyun.com/kubernetes/yum/repos/kubernetes-el7-x86_64
enabled=1
gpgcheck=0
repo_gpgcheck=0
gpgkey=https://mirrors.aliyun.com/kubernetes/yum/doc/yum-key.gpg https://mirrors.aliyun.com/kubernetes/yum/doc/rpm-package-key.gpg
EOF
bash
sudo tee /etc/yum.repos.d/kubernetes.repo > /dev/null <<EOF
[kubernetes]
name=Kubernetes
baseurl=https://mirrors.cloud.tencent.com/kubernetes/yum/repos/kubernetes-el7-x86_64
enabled=1
gpgcheck=0
EOF

查看可用版本 ​

bash
yum list kubeadm --showduplicates | grep kubernetes

下载离线包 ​

bash
yum install -y --downloadonly --downloaddir=rpm/kubeadm-rpm kubelet-1.28.2 kubeadm-1.28.2 kubectl-1.28.2 --disableexcludes=kubernetes

不生效刷新缓存 ​

bash
yum clean all && yum makecache

镜像处理 ​

bash
rpm -Uvh --force --nodeps rpm/docker-rpm/*.rpm
systemctl start docker
rpm -Uvh --force --nodeps rpm/kubeadm-rpm/*.rpm
systemctl enable --now kubelet
kubeadm config images pull --kubernetes-version 1.28.2 --image-repository registry.aliyuncs.com/google_containers --cri-socket=unix:///var/run/cri-dockerd.sock
docker save $(docker images | grep -v REPOSITORY | awk 'BEGIN{OFS=":";ORS=" "}{print $1,$2}') -o kube-1.28.2-img.tar
